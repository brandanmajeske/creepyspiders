package
{
    import flash.events.Event;
    import flash.events.MouseEvent;
    
    public class Spider extends SpiderBase
    {    
        //speedX is public to be accessible by other functions
        public var speedY:Number = 1 + Math.random() * 8;
        
        public function Spider()
        {
            super();
            //randomize the size of the spider
            this.scaleX = this.scaleY = .35 + Math.random() * 1;
            this.x =  50 + (Math.random() * 800);
            this.y = -100;
            //add event listener for moveSPider function
            this.addEventListener(Event.ENTER_FRAME, moveSpider);        
            
        }
        

            
        
        //function moveSpider - make spiders move right to left, if they go off the stage make them reappear on the right side again.
        private function moveSpider(event:Event):void
        {
            //rotate the spider to be facing left
            this.rotation = -180;
            //set width of spider to variable w
            var w:Number = this.height/2;
            this.y += speedY;
            
            if(this.y > 600 + w)
            {
                this.y = -50 - w;
            }
            
        }
            
    }
}
