package
{
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.display.StageDisplayState;
    
    [SWF("1000", height="600", backgroundColor="#ffbb00")]
    

    public class Main extends Sprite
    {

        public function Main()
        {
            addSpider();
        }
        
        private function addSpider():void
        {
            for (var i:uint = 0; i < 2; i++)
            {            
                var s:Spider = new Spider();
                this.addChild(s);
                s.addEventListener(MouseEvent.CLICK, removeSpider);
            }
        }
            
        private function removeSpider(event:MouseEvent):void
        {
                var spider:Spider = Spider(event.currentTarget);
                removeChild(spider);
                addSpider();
        }
    }
}
